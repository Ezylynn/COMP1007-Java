BEGIN Celsius
  MAIN:
    
    
    PRINT("Which temp scale are you converting from?")
    PRINT("(C)elsisus")
    PRINT("(F)arenheit")
    userInput = INPUT()

    IF (userInput == 'C') THEN
      tempInput = INPUT("Enter temp: ")
      faren = (tempInput*9/5)+32
      PRINT(faren)

    ELSE IF (userInput == 'F') THEN
      tempInput = INPUT("Enter temp: ")
      cel = (tempInput - 32) * 5/9
      PRINT(cel)

      ELSE THEN
        PRINT("Invalid Input. Terminating Program...")
  END MAIN
END
